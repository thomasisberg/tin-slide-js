Navigation

Opt out of non touch swipe navigation. Use the previous and next navigation buttons (nav), but not the dots.

TinSlide(document.getElementById('slider'), {
  swipe: {
    on: true,  // Default
    touchOnly: true
  },
  generate: {
    dots: {
      on: false
    },
    nav: {
      on: true // Default
    }
  }
});
